Generics

Een (extreem minimale cache) implementatie moet gedupliceerd worden voor elk Type dat men wenst te cachen.

Types
public class Person
{
    public int Id { get; set; }
    public string Email { get; set; }

    public override string ToString()
    {
        return $"Id: {Id}, Email: {Email}";
    }
}

public class Pet
{
    public int Id { get; set; }
    public string VetNotes { get; set; }
    public string Name { get; set; }
    public override string ToString(){
        return $"Id: {Id}, Name: {Name}, VetNotes: {VetNotes}";}
}
Cache
public class PersonCache
{
    private readonly Dictionary<int, Person> _dic = new();

    public void Add(int id, Person person)
    {
        _dic[id] = person;
    }

    public Person Get(int id)
    {
        return _dic[id];
    }
}

public class PetCache
{
    private readonly Dictionary<int, Pet> _dic = new();

    public void Add(int id, Pet pet)
    {
        _dic[id] = pet;
    }

    public Pet Get(int id)
    {
        return _dic[id];
    }
}
Cache Gebruik
var pCache = new PersonCache();
var person = new Person() { Id = 1, Email = "jefke@hotmail.com" };
pCache.Add(1, person);
Console.WriteLine(pCache.Get(1));

var petCache = new PetCache();
var pet = new Pet() { Id = 1, Name = "Misty",  VetNotes = "Some notes" };
petCache.Add(1, pet);
Console.WriteLine(petCache.Get(1));

Cache Generic
public class GenericCache<T>
{
    private readonly Dictionary<int, T> _dic = new();
    public void Add(int id, T pet)
    {
        _dic[id] = pet;
    }

    public T Get(int id)
    {
        return _dic[id];
    }
}

Een generieke methode veel doen, zolang de logica niet Type afhankelijk is.

Reverse Generic
public class GenericCache<T>
{
    //...

    public T[] ReverseArray(T[] arr)
    {
        var localArray = arr.ToArray();

        int leftIndex = 0;
        int rightIndex = localArray.Length - 1;
        T val;
        while (leftIndex < rightIndex)
        {
            val = localArray[leftIndex];
            localArray[leftIndex] = localArray[rightIndex];
            localArray[rightIndex] = val;
            leftIndex++;
            rightIndex--;
        }

        return localArray;
    }
}
Gebruik
var pCache = new GenericCache<Person>();
var person = new Person() { Id = 1, Email = "jefke@hotmail.com" };
pCache.Add(1, person);
Console.WriteLine(pCache.Get(1));

var petCache = new GenericCache<Pet>();
var pet = new Pet() { Id = 1, Name = "Misty",  VetNotes = "Some notes" };
petCache.Add(1, pet);
Console.WriteLine(petCache.Get(1));

Properties
public class Order
{
    public int Items { get; set; }
    public decimal Cost { get; set; }
}

public static class OrderManager
{
    public static decimal CalculateDiscount(Order order) =>
        order switch
        {
            { Items: > 10, Cost: > 1000.00m } => 0.10m,
            { Items: > 5, Cost: > 500.00m } => 0.05m,
            { Cost: > 250.00m } => 0.02m,
            null => throw new ArgumentNullException(nameof(order), "Can't calculate discount on null order"),
            var someObject => 0m,
        };
}

Union Types met OneOf

Exceptions in c# zijn "unchecked", dwz het afhandelen van exceptions wordt niet afgedwongen.

Los van andere redenen om geen Exceptions te gebruiken om business flows op te bouwen, heb je dus ook geen compile-time checks om te zien of alle scenario’s afgehandeld worden.

Oplossing

Meerdere return types per methode, 1 voor elk business scenario.

De meest gebruikte library om dit te doen is OneOf.

Bekijk een voorbeeld in de PG2 GitHub in het CreateCustomer pad.
